前言：grp是从Unix组数据库中读取组数据,通常是/etc/group

import grp
import operator  #主要是运算符操作模块
all_groups = grp.getgrall()  #得到所有系统的组信息
interesting_groups = sorted((g   #排序
                            for g in all_groups 
                            if not g.gr_name.startswith('_')),  #去掉排除'_'开头的组
                            key=operator.attrgetter('gr_name'))  #这个函数返回一个可调用对象,取此对象的attr属性值（tuple方式）



# Find the longest length for the name
name_length = max(len(g.gr_name) for g in interesting_groups) + 1



# Print report headers
fmt = '%-*s %4s %10s %s'
print fmt % (name_length, 'Name', 
             'GID', 
             'Password',
             'Members')
print '-' * name_length, '----', '-' * 10, '-' * 30



# Print the data
for g in interesting_groups:
    print fmt % (name_length, g.gr_name, 
                 g.gr_gid, 
                 g.gr_passwd,
                 ', '.join(g.gr_mem))

import functools



def show_details(name, f):
    """Show details of a callable object."""
    print '%s:' % name
    print '\tobject:', f
    print '\t__name__:', 
    try:
        print f.__name__
    except AttributeError:
        print '(no __name__)'
    print '\t__doc__', repr(f.__doc__)
    print
    return



def simple_decorator(f):
    @functools.wraps(f)
    def decorated(a='decorated defaults', b=1):
        print '\tdecorated:', (a, b)
        print '\t',
        f(a, b=b)
        return
    return decorated



def myfunc(a, b=2):
    print '\tmyfunc:', (a,b)
    return



show_details('myfunc', myfunc)
myfunc('unwrapped, default b')
myfunc('unwrapped, passing b', 3)
print

show_details('wrapped_myfunc', wrapped_myfunc)

关于cStringIO研究

Jun 12th, 2012

前言：cStringIO的行为与file对象非常像，但它不是磁盘上文件，而是一个内存里的“文件”，我们可以将操作磁盘文件那样来操作cStringIO，经常被用来作为字符串的缓存，因为StringIO有个好处，他的有些接口和文件操作是一致的

#coding=utf-8
try:
    from cStringIO import StringIO
except:
    from StringIO import StringIO



output = StringIO()
output.write('This goes into the buffer. ') #写到内存缓存里面
print >>output, 'And so does this.'
print output.getvalue() #找到缓存结果
output.close() #放弃内存中的缓存



input = StringIO('Inital value for read buffer') #初始化一个缓存区



print input.read() # 读取缓存区内容

dongwm@linux-dongwm:~> python test.py This goes into the buffer. And so does this.

关于cPickle研究

Jun 12th, 2012

前言：cPickle是一个让python对象序列化的模块，什么是序列化？比如内存里面有个数据结构想保存，重用甚至发送给别人（比如游戏保存配置，下次登录游戏还是在上次的那个场景的意思一样），为什么这里只说cPickle？它是Pickle在python中以C的形式实现，性能快很多倍（算法其实一样），但是假如在进程间通信或者存储，数据是不安全的

try:
    import cPickle as pickle
except:
    import pickle
import pprint #这个模块将数据格式化显示了



data1 = [ { 'a':'A', 'b':2, 'c':3.0 } ]
print 'BEFORE:',
pprint.pprint(data1)



data1_string = pickle.dumps(data1)  #使用dump编码 类似于json
pprint.pprint(data1_string)
data2 = pickle.loads(data1_string)  #使用load解码
print 'AFTER:',
pprint.pprint(data2)



print 'SAME?:', (data1 is data2)
print 'EQUAL?:', (data1 == data2)

 

执行结果：

dongwm@linux-dongwm:~> python test.py BEFORE:[{‘a’: ‘A’, ‘b’: 2, ‘c’: 3.0}]

“(lp1\n(dp2\nS’a’\nS’A’\nsS’c’\nF3\nsS’b’\nI2\nsa.” AFTER:[{‘a’: ‘A’, ‘b’: 2, ‘c’: 3.0}] SAME?: False  #前后数据相同，但是不占用同样的内存位置 EQUAL?: True 

try:
    import cPickle as pickle
except:
    import pickle
import pprint
from StringIO import StringIO



class SimpleObject(object):



    def __init__(self, name):
        self.name = name
        l = list(name)
        l.reverse()
        self.name_backwards = ''.join(l)
        return



data = []
data.append(SimpleObject('pickle')) #添加多个对象  
data.append(SimpleObject('cPickle'))
data.append(SimpleObject('last'))



out_s = StringIO() 



for o in data:
    print 'WRITING: %s (%s)' % (o.name, o.name_backwards)
    pickle.dump(o, out_s)
    out_s.flush()



in_s = StringIO(out_s.getvalue()) #写入流中多个对象



while True:
    try:
        o = pickle.load(in_s) #从流中读取，而不需要事先知道对象多大多少
    except EOFError:
        break
    else:
        print 'READ: %s (%s)' % (o.name, o.name_backwards)

执行过程：

dongwm@linux-dongwm:~> python test.py WRITING: pickle (elkcip) WRITING: cPickle (elkciPc) WRITING: last (tsal) READ: pickle (elkcip) READ: cPickle (elkciPc) READ: last (tsal)
