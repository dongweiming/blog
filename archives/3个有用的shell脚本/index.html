3个有用的shell脚本

Mar 4th, 2011

1  监视系统的CPU，内存使用等情况：（根据CPU和内存使用排序） 

$ cat processes.sh
#! /bin/bash
#用于显示CPU使用和内存使用的检查



echo "开始时间" `date`
if [ $# -eq 0 ]
then



	echo "CPU使用率升序排列"
	ps -e -o pcpu,cpu,nice,state,cputime,args --sort pcpu  # sorted based on %cpu
	echo "内存使用升序排列"
	ps -e -orss=,args= | sort -b -k1,1n # sorted bases rss value



# 可以添加位置参数(mem/cpu)
else
	case "$1" in
	mem)
	 echo "内存使用升序排列"
 	 ps -e -orss=,args= | sort -b -k1,1n
	 ;;
 	cpu)
	 echo "CPU使用率升序排列"
	 ps -e -o pcpu,cpu,nice,state,cputime,args --sort pcpu
	 ;;
 	*)
		echo "请输入正确的参数 \n"
		echo "可以使用 : $0 mem/cpu"
		exit 1
 	esac	



fi
echo "结束时间" `date`
exit 0
使用方式： 
$ processes.sh
$ processes.sh mem
$ processes.sh cpu
